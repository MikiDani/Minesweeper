CREATE TABLE score (
    id int NOT NULL PRIMARY KEY,
    userid int,
    recordtime text,
    recorddate text,
    tablesize text,
    bombs text,
    PRIMARY KEY (id),
    FOREIGN KEY (userid) REFERENCES users(id)
);


SELECT nickname, recordtime, tablesize, bombs, recorddate FROM score
INNER JOIN users ON (score.userid = users.id)
ORDER BY nickname DESC
